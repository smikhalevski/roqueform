Interface ErrorsPlugin<Error>

The plugin that enables a field errors.

interface ErrorsPlugin<Error> {
    errors: readonly Error[];
    isInvalid: boolean;
    addError(error: Error): void;
    clearErrors(options?: ClearErrorsOptions): void;
    deleteError(error: Error): void;
    getInvalidFields(): BareField<any, any>[];
    on(eventType: "change:errors", subscriber: Subscriber<Error[], any>): Unsubscribe;
}
